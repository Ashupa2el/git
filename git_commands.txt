### Switching the Branch in Git ### A magic moment ###

# --------------------------------------------------
# STEP 1: Go to Coding directory
# --------------------------------------------------
cd "D:\Ashu\OneDrive\Desktop\The Directory\Coding"


# --------------------------------------------------
# STEP 2: Enter your Git project folder
# (example: git)
# --------------------------------------------------
cd git


# --------------------------------------------------
# STEP 3: Check current Git state
# Shows branch, cleanliness, changes
# --------------------------------------------------
git status


# --------------------------------------------------
# STEP 4: Create a new branch
# This creates a new POINTER at current commit
# Does NOT switch to it
# --------------------------------------------------
git branch feature-1


# --------------------------------------------------
# STEP 5: Switch to the new branch (THE MAGIC)
# HEAD now points to feature-1
# Files change to match that branch's commit
# --------------------------------------------------
git switch feature-1
# (older Git: git checkout feature-1)


# --------------------------------------------------
# STEP 6: Verify current branch
# * indicates the active branch
# --------------------------------------------------
git branch


# --------------------------------------------------
# STEP 7: Make changes on this branch
# (example file edit)
# --------------------------------------------------
notepad test.txt
# add text, save, close


# --------------------------------------------------
# STEP 8: Check status on this branch
# Changes exist ONLY on feature-1
# --------------------------------------------------
git status


# --------------------------------------------------
# STEP 9: Stage changes
# Select what will go into the commit
# --------------------------------------------------
git add test.txt
# or: git add .


# --------------------------------------------------
# STEP 10: Commit changes on feature-1
# Branch pointer moves forward
# --------------------------------------------------
git commit -m "Update test.txt on feature-1"


# --------------------------------------------------
# STEP 11: Switch back to main branch
# Files revert to main branch snapshot
# (THIS IS THE WOW MOMENT)
# --------------------------------------------------
git switch main


# --------------------------------------------------
# STEP 12: Open file again to observe difference
# You will NOT see feature-1 changes here
# --------------------------------------------------
notepad test.txt


# --------------------------------------------------
# STEP 13: Mental model reminder
# --------------------------------------------------
# - Commits store files
# - Branches point to commits
# - Switching branches = switching commits
# - Git rewrites files safely
